**src/pages/index.astro** (DE):
```astro
---
import BaseLayout from '../layouts/BaseLayout.astro'
import PageBuilder from '../components/blocks/PageBuilder.astro'
import { getHomepageWithSections, getSiteSettings } from '../lib/sanity'

const locale = 'de' as const
const page = await getHomepageWithSections(locale)
const settings = await getSiteSettings()
const sections = page?.sections ?? []
const title = page?.seo?.title ?? settings?.seo?.title ?? settings?.siteName ?? 'nestorsegura.com'
const description = page?.seo?.description ?? settings?.seo?.description ?? settings?.tagline
---
<BaseLayout {title} {locale} {description}>
  <PageBuilder {sections} {locale} />
</BaseLayout>
```

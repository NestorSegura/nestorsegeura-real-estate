```astro
---
import type { GetStaticPaths } from 'astro'
import BaseLayout from '../../layouts/BaseLayout.astro'
import PTRenderer from '../../components/portabletext/PTRenderer.astro'
import BlogTOC from '../../components/blog/BlogTOC.astro'
import { getAllPostSlugsForLocale, getFullPostForLocale, getSiteSettings } from '../../lib/sanity'
import { extractToc } from '../../lib/tocFromPT'
import { readingTime } from '../../lib/readingTime'
import { urlFor } from '../../lib/imageUrl'

const locale = 'de' as const

export const getStaticPaths = (async () => {
  const slugs = await getAllPostSlugsForLocale(locale)
  return slugs.map(({ slug }) => ({ params: { slug } }))
}) satisfies GetStaticPaths

const { slug } = Astro.params
const post = await getFullPostForLocale(slug!, locale)
if (!post) return Astro.redirect('/404')

const settings = await getSiteSettings()
const toc = extractToc(post.body ?? [])
const minutes = readingTime(post.body ?? [], locale)
const dateLocale = locale === 'de' ? 'de-DE' : locale === 'es' ? 'es-ES' : 'en-US'
const dateFmt = new Intl.DateTimeFormat(dateLocale, { dateStyle: 'long' }).format(new Date(post.publishedAt))

const title = post.seo?.title ?? post.title
const description = post.seo?.description ?? post.excerpt
---
<BaseLayout {title} {locale} {description}>
  <article class="container mx-auto px-4 py-12 max-w-6xl">
    <header class="mb-8 max-w-3xl">
      <h1 class="text-4xl lg:text-5xl mb-4">{post.title}</h1>
      <div class="flex items-center gap-3 text-sm text-gray-600">
        {post.author?.image && (
          <img src={urlFor(post.author.image)?.width(80).format('webp').url()} alt={post.author.name} width="40" height="40" class="rounded-full" loading="lazy" />
        )}
        <span>{post.author?.name}</span>
        <span>·</span>
        <time datetime={post.publishedAt}>{dateFmt}</time>
        <span>·</span>
        <span>{minutes} min</span>
      </div>
      {post.tags?.length > 0 && (
        <div class="flex gap-2 mt-3">
          {post.tags.map(tag => <span class="px-2 py-1 text-xs bg-primary-100 text-primary-700 rounded">{tag}</span>)}
        </div>
      )}
    </header>
    {post.mainImage && (
      <img src={urlFor(post.mainImage)?.width(1280).format('webp').url()} alt={post.mainImage.alt ?? ''} class="w-full rounded-lg mb-12" loading="eager" width="1280" height="720" />
    )}
    <div class="grid gap-12 lg:grid-cols-[1fr_240px]">
      <div class="prose max-w-none">
        <PTRenderer value={post.body ?? []} />
      </div>
      <aside><BlogTOC items={toc} {locale} /></aside>
    </div>
  </article>
</BaseLayout>
```

Adjust import paths for /en/ and /es/ subdirs. `getAllPostSlugsForLocale` is added in Plan 08-01 Task 4 — import it directly from `src/lib/sanity`; do NOT add it here.

`category` field rendering: optional, skip for v1 if not in post schema (verify in src/sanity/schemas/documents/post.ts).

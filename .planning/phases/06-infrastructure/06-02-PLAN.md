    1. Update src/pages/index.astro to import and use the Sanity client:
       ```astro
       ---
       import '../styles/global.css';
       import { sanityClient } from "sanity:client";

       let sanityStatus = "connecting...";
       let siteName = "";
       let fetchError = "";

       try {
         const settings = await sanityClient.fetch(
           `*[_type == "siteSettings"][0]{ siteName, siteDescription }`
         );
         if (settings) {
           sanityStatus = "OK";
           siteName = settings.siteName ?? "(no siteName field)";
         } else {
           sanityStatus = "Connected but no siteSettings document found";
         }
       } catch (e) {
         sanityStatus = "FAILED";
         fetchError = e instanceof Error ? e.message : String(e);
         console.error("Sanity fetch failed:", e);
       }
       ---
       <!doctype html>
       <html lang="de">
         <head>
           <meta charset="UTF-8" />
           <meta name="viewport" content="width=device-width, initial-scale=1.0" />
           <title>{siteName || 'Astro scaffold'}</title>
         </head>
         <body class="bg-white text-gray-900 p-8 max-w-2xl mx-auto">
           <h1 class="text-3xl font-bold">Astro 6 + Cloudflare Workers</h1>
           <div class="mt-6 p-4 rounded border">
             <h2 class="text-lg font-semibold">Infrastructure Status</h2>
             <ul class="mt-2 space-y-1 text-sm">
               <li>Astro: <span class="text-green-600 font-mono">OK</span></li>
               <li>Sanity: <span class={sanityStatus === "OK" ? "text-green-600 font-mono" : "text-red-600 font-mono"}>{sanityStatus}</span></li>
               {siteName && <li>Site name: <span class="font-mono">{siteName}</span></li>}
               {fetchError && <li class="text-red-600">Error: {fetchError}</li>}
             </ul>
           </div>
           <p class="mt-6 text-sm text-gray-500">Phase 6 scaffold complete. No pages, no components — just foundation.</p>
         </body>
       </html>
       ```

    4. Verify dist/_worker.js/index.js still exists after build.

    5. Run `npx wrangler dev` and confirm the built page shows Sanity data.
